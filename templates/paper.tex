\newcommand{\atable}[4]{
    \ifthenelse{\equal{#4}{wide}}{ 
        \begin{table*}[htbp]
            \centering
            \caption{#2}
            \label{tab:#3}
            \pgfplotstabletypeset[
                col sep=comma,
                string type,
                every head row/.style={before row=\toprule,after row=\midrule},
                every last row/.style={after row=\bottomrule},
                header=true,
                begin table=\begin{tabular},
                end table=\end{tabular},
            ]{#1}
        \end{table*}

    }{

        \begin{minipage}{\columnwidth}
        \begin{table}[H]
            \centering
            \caption{#2}
            \label{tab:#3}
            \pgfplotstabletypeset[
                col sep=comma,
                string type,
                every head row/.style={before row=\toprule,after row=\midrule},
                every last row/.style={after row=\bottomrule},
                header=true,
                begin table=\begin{tabular},
                end table=\end{tabular},
            ]{#1}
        \end{table}
        \end{minipage}
    }

}
